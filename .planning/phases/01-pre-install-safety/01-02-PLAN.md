    ```js
    export async function startStubServer({ mode = 'healthy', delayMs = 0 } = {}) {
      const captured = { lastAuthHeader: null, lastBody: null, lastUrl: null };
      const state = { mode, delayMs };

      const server = http.createServer(async (req, res) => {
        captured.lastAuthHeader = req.headers.authorization ?? null;
        captured.lastUrl = req.url;
        const chunks = [];
        for await (const c of req) chunks.push(c);
        captured.lastBody = chunks.length ? Buffer.concat(chunks).toString('utf8') : null;

        if (state.delayMs > 0) await new Promise(r => setTimeout(r, state.delayMs));

        if (state.mode === 'unauth' && req.headers.authorization === undefined) {
          res.writeHead(401, { 'content-type': 'application/json' });
          return res.end(JSON.stringify({ error: 'unauthorized' }));
        }
        if (state.mode === 'down') {
          res.writeHead(500, { 'content-type': 'application/json' });
          return res.end(JSON.stringify({ error: 'internal' }));
        }

        const respond = (status, body) => {
          res.writeHead(status, { 'content-type': 'application/json' });
          res.end(JSON.stringify(body));
        };

        if (req.url === '/health') return respond(200, { ok: true, paperclipVersion: '0.41.2' });
        if (req.url.startsWith('/api/issues')) return respond(200, [{ id: 'BEAAA-1', title: 'Sample' }]);
        if (req.url.includes('/agents') && !req.url.includes('/heartbeat/')) return respond(200, [{ id: 'agent-foo', role: 'editor' }]);
    ```

    ```js
    function buildHeaders(apiKey) {
      return apiKey ? { 'authorization': `Bearer ${apiKey}` } : {};
    }

    async function call(opts, method, pathname, body) {
      const url = `${opts.apiUrl.replace(/\\/$/, '')}${pathname}`;
      const init = { method, headers: { ...buildHeaders(opts.apiKey) }, signal: opts.signal };
      if (body !== undefined) {
        init.headers['content-type'] = 'application/json';
        init.body = JSON.stringify(body);
      }
      try {
        const raw = await fetch(url, init);
        const ct = raw.headers.get('content-type') ?? '';
        const body = ct.includes('application/json') ? await raw.json() : await raw.text();
        return { status: raw.status, body, raw };
      } catch (err) {
        throw redactedError(err, opts.apiKey);
      }
    }

    export function getHealth(opts) { return call(opts, 'GET', '/health'); }
    export function listIssues(opts, q = {}) { return call(opts, 'GET', `/api/issues?limit=${q.limit ?? 1}`); }
    export function listCompanyAgents(opts, companyId) { return call(opts, 'GET', `/api/companies/${encodeURIComponent(companyId)}/agents`); }
    export function listPlugins(opts) { return call(opts, 'GET', '/api/plugins'); }
    export function invokeHeartbeat(opts, agentId, payload = {}) { return call(opts, 'POST', `/api/agents/${encodeURIComponent(agentId)}/heartbeat/invoke`, payload); }

    export function redactedError(err, apiKey) {
      if (!apiKey) return err;
      const msg = (err.message ?? '').split(apiKey).join('<REDACTED>');
      const wrapped = new Error(msg);
      wrapped.name = err.name;
      wrapped.cause = err.cause;
      return wrapped;
    }
    ```

    ```js
    import * as api from './paperclip-api.mjs';
    import { readManifest } from './manifest.mjs';
    import path from 'node:path';

    const DEFAULT_TIMEOUT = 5000;

    /** Compose per-check timeout with optional outer deadline.
     *  Returns a signal that aborts when EITHER fires (whichever is first wins).
     *  When the outer deadline aborts, the result is tagged so the caller can label
     *  the failure 'rehearsal time exceeded' instead of a generic timeout. */
    function makeSignal(timeoutMs, deadline) {
      const localCtrl = new AbortController();
      const t = setTimeout(() => localCtrl.abort(new Error('per-check-timeout')), timeoutMs);

      let composed;
      if (deadline) {
        // Node ≥20.3 — AbortSignal.any handles the OR-merge.
        composed = AbortSignal.any([localCtrl.signal, deadline]);
      } else {
        composed = localCtrl.signal;
      }

      return {
        signal: composed,
        cleanup: () => clearTimeout(t),
        deadlineFired: () => deadline?.aborted === true,
      };
    }

    export async function smoke(opts) {
      const { apiUrl, apiKey, companyId, editorAgentId, timeoutMs = DEFAULT_TIMEOUT, snapshotId, snapshotsDir, deadline } = opts;
      const checks = [];
      const apiOpts = (signal) => ({ apiUrl, apiKey, signal });

      let healthBody = null;
      let pluginsBody = null;

      // 1. health
      {
        const { signal, cleanup, deadlineFired } = makeSignal(timeoutMs, deadline);
        try {
          const res = await api.getHealth(apiOpts(signal));
          if (res.status >= 500) { checks.push({ name: 'health', status: 'fail', detail: `HTTP ${res.status}` }); return finalize(checks); }
          healthBody = res.body;
          checks.push({ name: 'health', status: 'pass' });
        } catch (e) {
          const reason = deadlineFired() ? 'rehearsal time exceeded' : e.message;
          checks.push({ name: 'health', status: 'fail', detail: reason });
          return finalize(checks);
        } finally { cleanup(); }
      }

      // 2. issues
      {
        const { signal, cleanup, deadlineFired } = makeSignal(timeoutMs, deadline);
        try {
          const res = await api.listIssues(apiOpts(signal), { limit: 1 });
          if (res.status >= 500 || !Array.isArray(res.body)) { checks.push({ name: 'issues', status: 'fail', detail: `HTTP ${res.status}` }); return finalize(checks); }
          checks.push({ name: 'issues', status: 'pass' });
        } catch (e) {
          const reason = deadlineFired() ? 'rehearsal time exceeded' : e.message;
          checks.push({ name: 'issues', status: 'fail', detail: reason });
          return finalize(checks);
        } finally { cleanup(); }
      }

      // 3. agents
      {
        const { signal, cleanup, deadlineFired } = makeSignal(timeoutMs, deadline);
        try {
          const res = await api.listCompanyAgents(apiOpts(signal), companyId);
          if (res.status >= 500 || !Array.isArray(res.body)) { checks.push({ name: 'agents', status: 'fail', detail: `HTTP ${res.status}` }); return finalize(checks); }
          checks.push({ name: 'agents', status: 'pass' });
        } catch (e) {
          const reason = deadlineFired() ? 'rehearsal time exceeded' : e.message;
          checks.push({ name: 'agents', status: 'fail', detail: reason });
          return finalize(checks);
        } finally { cleanup(); }
      }

      // 4. plugins
      {
        const { signal, cleanup, deadlineFired } = makeSignal(timeoutMs, deadline);
        try {
          const res = await api.listPlugins(apiOpts(signal));
          if (res.status >= 500 || !Array.isArray(res.body)) { checks.push({ name: 'plugins', status: 'fail', detail: `HTTP ${res.status}` }); return finalize(checks); }
          pluginsBody = res.body;
          checks.push({ name: 'plugins', status: 'pass' });
        } catch (e) {
          const reason = deadlineFired() ? 'rehearsal time exceeded' : e.message;
          checks.push({ name: 'plugins', status: 'fail', detail: reason });
          return finalize(checks);
        } finally { cleanup(); }
      }

      // 5. heartbeat — 4xx is acceptable per research Open Question 2
      if (!editorAgentId) {
        checks.push({ name: 'heartbeat', status: 'skipped', detail: 'no editor-agent id' });
      } else {
        const { signal, cleanup, deadlineFired } = makeSignal(timeoutMs, deadline);
        try {
          const res = await api.invokeHeartbeat(apiOpts(signal), editorAgentId, {});
          if (res.status >= 500) { checks.push({ name: 'heartbeat', status: 'fail', detail: `HTTP ${res.status}` }); return finalize(checks); }
          if (res.status >= 400) checks.push({ name: 'heartbeat', status: 'pass', detail: `http-${res.status}-server-alive` });
          else checks.push({ name: 'heartbeat', status: 'pass' });
        } catch (e) {
          const reason = deadlineFired() ? 'rehearsal time exceeded' : e.message;
          checks.push({ name: 'heartbeat', status: 'fail', detail: reason });
          return finalize(checks);
        } finally { cleanup(); }
      }

      // 6. cross-check vs manifest if snapshotId provided
      if (snapshotId && snapshotsDir) {
        const snapshotDir = path.join(snapshotsDir, snapshotId);
        const manifest = await readManifest(snapshotDir);

        // plugin set equality (always required when snapshotId provided)
        const expected = new Set((manifest.installedPlugins ?? []).map(p => p.id));
        const actual = new Set((pluginsBody ?? []).map(p => p.id));
        const missing = [...expected].filter(x => !actual.has(x));
        const extra = [...actual].filter(x => !expected.has(x));
        if (missing.length || extra.length) {
          checks.push({ name: 'plugin-list-cross-check', status: 'fail', detail: `missing=[${missing.join(',')}] extra=[${extra.join(',')}]` });
          return finalize(checks);
        }
        checks.push({ name: 'plugin-list-cross-check', status: 'pass' });

        // version equality — conditional on the server reporting a version.
        // When /health body lacks paperclipVersion we cannot enforce; status: skipped.
        const restoredVersion = healthBody && typeof healthBody === 'object' ? healthBody.paperclipVersion : undefined;
        if (restoredVersion && restoredVersion !== manifest.paperclipVersion) {
          checks.push({ name: 'version-cross-check', status: 'fail', detail: `expected ${manifest.paperclipVersion} got ${restoredVersion}` });
          return finalize(checks);
        }
        checks.push({
          name: 'version-cross-check',
          status: restoredVersion ? 'pass' : 'skipped',
          detail: restoredVersion ? undefined : 'server did not report paperclipVersion in /health body; version equality not enforced',
        });
      }

      return finalize(checks);
    }

    function finalize(checks) {
      const failed = checks.find(c => c.status === 'fail');
      if (failed) return { ok: false, checks, failedCheck: failed.name, reason: failed.detail };
      return { ok: true, checks };
    }
    ```
