export interface GateOpts {
  maxAgeMinutes?: number;        // default 15 (SAFE-05 default)
  snapshotsDir: string;          // root of .planning/snapshots/
  innerCommand: string[];        // ['pnpm', 'paperclipai', 'plugin', 'install', 'clarity-pack']

export interface GateResult {
  forwarded: boolean;            // true if inner command was spawned

    import path from 'node:path';
    import { promises as fs } from 'node:fs';
    import spawn from 'cross-spawn';
    import { listSnapshots } from './list.mjs';
    import { readManifest } from './manifest.mjs';

    const BYPASS_ENV_FRESHNESS_MS = 60_000;

    export function isFreshAndVerified(manifest, maxAgeMinutes, now = new Date()) {
      if (!manifest.verifiedAt) return { ok: false, reason: 'snapshot-not-verified' };
      const verifiedMs = Date.parse(manifest.verifiedAt);
      const ageMs = now.getTime() - verifiedMs;
      if (ageMs > maxAgeMinutes * 60_000) return { ok: false, reason: 'snapshot-stale' };
      return { ok: true };
    }

    export async function findLatestSnapshot(snapshotsDir) {
      try {
        const items = await listSnapshots(snapshotsDir);
        if (!items.length) return null;
        const newest = items[0];
        const manifest = await readManifest(path.join(snapshotsDir, newest.id));
        return { snapshotId: newest.id, manifest };
      } catch (err) {
        if (err.code === 'ENOENT') return null;
        throw err;
      }
    }

    export function checkBypassEnv(env, now = new Date()) {
      const v = env.CLARITY_SAFETY_BYPASS;
      if (!v) return { allowed: false };
      const m = /^I_KNOW=(\d+)$/.exec(v);
      if (!m) return { allowed: false, reason: 'bypass-env-malformed' };
      const ts = parseInt(m[1], 10);
      if (Number.isNaN(ts)) return { allowed: false, reason: 'bypass-env-malformed' };
      const age = now.getTime() - ts;
      if (age < 0 || age > BYPASS_ENV_FRESHNESS_MS) return { allowed: false, reason: 'bypass-env-stale' };
      return { allowed: true };
    }

    async function logBypass(rehearsalLogPath, innerCommand, reason) {
      const line = `\n[BYPASS] ${new Date().toISOString()} cmd=${innerCommand.join(' ')} reason=${reason ?? ''}\n`;
      try {
        await fs.appendFile(rehearsalLogPath, line, 'utf8');
      } catch (err) {
        process.stderr.write(`warning: bypass log write failed (${err.message}); event: ${line}`);
      }
    }

    export async function gate(opts) {
      const {
        maxAgeMinutes = 15,
        snapshotsDir,
        innerCommand,
        env = process.env,
        rehearsalLogPath = path.join(process.cwd(), 'runbook', 'REHEARSAL.md'),
        _spawn = spawn,
      } = opts;

      if (!innerCommand || innerCommand.length === 0) {
        throw new Error('gate requires innerCommand to be a non-empty array');
      }

      const wantsBypass = innerCommand.includes('--gate-bypass');
      const bypassCheck = wantsBypass ? checkBypassEnv(env) : { allowed: false };
      if (wantsBypass && bypassCheck.allowed) {
        await logBypass(rehearsalLogPath, innerCommand, 'CLARITY_SAFETY_BYPASS valid');
        return runInner(_spawn, innerCommand, /* bypassed */ true);
      }
      if (wantsBypass && !bypassCheck.allowed) {
        return {
          forwarded: false,
          exitCode: null,
          refusalReason: 'snapshot-not-verified',
          remediation: `--gate-bypass requested but CLARITY_SAFETY_BYPASS=I_KNOW=<unix-ms-within-last-60s> is missing or invalid (${bypassCheck.reason ?? 'unknown'}).`,
        };
      }

      let latest;
      try {
        latest = await findLatestSnapshot(snapshotsDir);
      } catch (err) {
        return {
          forwarded: false,
          exitCode: null,
          refusalReason: 'manifest-unreadable',
          remediation: `Latest snapshot manifest could not be read: ${err.message}. Take a fresh snapshot:\n  pnpm clarity-safety snapshot && pnpm clarity-safety verify <new-snapshot-id>`,
        };
      }

      if (!latest) {
        return {
          forwarded: false,
          exitCode: null,
          refusalReason: 'no-snapshot',
          remediation: 'No snapshot found. Run:\n  pnpm clarity-safety snapshot\n  pnpm clarity-safety verify <new-snapshot-id>',
        };
      }

      const fresh = isFreshAndVerified(latest.manifest, maxAgeMinutes);
      if (!fresh.ok) {
        const remediation =
          fresh.reason === 'snapshot-not-verified'
            ? `Latest snapshot ${latest.snapshotId} is unverified. Run:\n  pnpm clarity-safety verify ${latest.snapshotId}`
            : `Latest snapshot ${latest.snapshotId} is older than ${maxAgeMinutes} minutes. Run:\n  pnpm clarity-safety snapshot\n  pnpm clarity-safety verify <new-snapshot-id>`;
        return { forwarded: false, exitCode: null, refusalReason: fresh.reason, remediation };
      }

      return runInner(_spawn, innerCommand, /* bypassed */ false);
    }

    function runInner(_spawn, innerCommand, bypassed) {
      return new Promise((resolve, reject) => {
        const [cmd, ...args] = innerCommand;
        const child = _spawn(cmd, args, { shell: false, stdio: 'inherit' });
        child.on('error', reject);
        child.on('exit', (code) => resolve({ forwarded: true, exitCode: code, bypassed }));
      });
    }
