// Discriminated error kinds — usePoll distinguishes transient (back off) from terminal (stop).
export type PollErrorKind =
  | 'WORKER_UNAVAILABLE'   // transient 5xx / worker restart in flight → exponential backoff
  | 'PLUGIN_DISABLED'      // HTTP 404 from host OR SDK error code 'PLUGIN_DISABLED' → TERMINAL, stop polling
  | 'TIMEOUT'              // transient → backoff
  | 'UNKNOWN';             // unexpected; treat as transient

export type PollError = { kind: PollErrorKind; status?: number; message?: string };

export type UsePollOptions<T> = {
  key: string;
  fetcher: () => Promise<T>;
  intervalMs: number;
  dedupeBy?: "content-hash" | "off";
  pauseOnHidden?: boolean;     // default true
  onWorkerUnavailable?: "exponential-backoff" | "stop";
};

export function usePoll<T>(opts: UsePollOptions<T>): {
  data: T | null;
  error: PollError | null;     // typed kind allows surfaces to render terminal vs transient UX
  stale: boolean;
  isLeader: boolean | null;    // null = leader-election unknown / not wired (02-04 wraps with leader)
};

       export async function resolveRefs(ids, fetcher) {
         if (ids.length === 0) return [];
         const unique = Array.from(new Set(ids));
         const fetched = await fetcher(unique);  // ONE round trip — PRIM-01
         const byId = new Map(fetched.map(r => [r.id, r]));
         return ids.map(id => {
           const r = byId.get(id);
           if (!r) return { id, title: 'unknown', status: 'unknown', ownerUserId: null, excerpt: null, url: '' };
           return {
             id: r.id, title: r.title, status: r.status, ownerUserId: r.ownerUserId,
             excerpt: r.bodyExcerptForViewer,  // PRIM-02 — null means permission denied

         });

        ```typescript
        // Murmur3 32-bit, synchronous. Public-domain reference implementation.
        // ≤50 LOC; no deps; usable in browser and Node 20+.
        function murmur3_32(input: string, seed = 0): string {
          let h1 = seed | 0;
          const remainder = input.length & 3;
          const bytes = input.length - remainder;
          const c1 = 0xcc9e2d51, c2 = 0x1b873593;
          let i = 0;
          while (i < bytes) {
            let k1 =
              (input.charCodeAt(i) & 0xff) |
              ((input.charCodeAt(i + 1) & 0xff) << 8) |
              ((input.charCodeAt(i + 2) & 0xff) << 16) |
              ((input.charCodeAt(i + 3) & 0xff) << 24);
            i += 4;
            k1 = Math.imul(k1, c1);
            k1 = (k1 << 15) | (k1 >>> 17);
            k1 = Math.imul(k1, c2);
            h1 ^= k1;
            h1 = (h1 << 13) | (h1 >>> 19);
            h1 = (Math.imul(h1, 5) + 0xe6546b64) | 0;
          }
          let k1 = 0;
          switch (remainder) {
            case 3: k1 ^= (input.charCodeAt(i + 2) & 0xff) << 16;
            // falls through
            case 2: k1 ^= (input.charCodeAt(i + 1) & 0xff) << 8;
            // falls through
            case 1:
              k1 ^= input.charCodeAt(i) & 0xff;
              k1 = Math.imul(k1, c1);
              k1 = (k1 << 15) | (k1 >>> 17);
              k1 = Math.imul(k1, c2);
              h1 ^= k1;
          }
          h1 ^= input.length;
          h1 ^= h1 >>> 16; h1 = Math.imul(h1, 0x85ebca6b);
          h1 ^= h1 >>> 13; h1 = Math.imul(h1, 0xc2b2ae35);
          h1 ^= h1 >>> 16;
          return (h1 >>> 0).toString(16).padStart(8, '0');
        }
        ```
        Use as: `const hash = murmur3_32(JSON.stringify(data));` (synchronous; one expression; no await). Compare with previous tick's hash before calling setState. No new npm dep, no async surface, no race condition.

        **(b) Lifecycle-aware error discrimination (revision warning #5; PITFALLS.md #1):**

        ```typescript
        function classifyFetchError(err: any): PollError {
          // SDK error shape (preferred — usePluginData rejects with structured errors)
          if (err?.code === 'PLUGIN_DISABLED') return { kind: 'PLUGIN_DISABLED', message: err.message };
          if (err?.code === 'WORKER_UNAVAILABLE') return { kind: 'WORKER_UNAVAILABLE', message: err.message };
          // HTTP-style fallback (in case the fetcher surfaces a Response-like object or status)
          const status: number | undefined = err?.status ?? err?.response?.status;
          if (status === 404) return { kind: 'PLUGIN_DISABLED', status, message: '404 — plugin disabled or uninstalled' };
          if (status === 503 || status === 502) return { kind: 'WORKER_UNAVAILABLE', status, message: `${status} transient` };
          if (err?.name === 'AbortError' || /timeout/i.test(String(err?.message))) return { kind: 'TIMEOUT', message: String(err?.message) };
          return { kind: 'UNKNOWN', message: String(err?.message ?? err) };
        }
        ```

        Polling state machine (pseudo-code; implement faithfully):
        ```typescript
        let timer: ReturnType<typeof setTimeout> | null = null;
        let stopped = false;       // PLUGIN_DISABLED sets this true; gates all further scheduling
        let backoffMs = intervalMs;

        async function tick() {
          if (stopped) return;
          if (pauseOnHidden && typeof document !== 'undefined' && document.visibilityState !== 'visible') {
            timer = setTimeout(tick, intervalMs);
            return;
          }
          try {
            const next = await fetcher();
            const hash = murmur3_32(JSON.stringify(next));  // synchronous
            if (dedupeBy !== 'off' && hash === lastHash) {
              // no setState — Pitfall #7 mitigation
            } else {
              lastHash = hash;
              setState({ data: next, error: null });
            }
            backoffMs = intervalMs;  // reset on success
            setError(null);
            timer = setTimeout(tick, intervalMs);
          } catch (e) {
            const pe = classifyFetchError(e);
            setError(pe);
            if (pe.kind === 'PLUGIN_DISABLED') {
              stopped = true;
              if (timer) { clearTimeout(timer); timer = null; }
              return;                                   // TERMINAL — no further scheduling
            }
            // transient: WORKER_UNAVAILABLE / TIMEOUT / UNKNOWN → exponential backoff
            backoffMs = Math.min(backoffMs * 2, 5 * 60_000);
            timer = setTimeout(tick, backoffMs);
          }
        }
        ```
