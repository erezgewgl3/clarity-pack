    3. **src/worker/agents/circuit-breaker.ts**:
       ```typescript
       export const MAX_CONSECUTIVE_FAILURES = 3;
       const counters = new Map<string, number>();
       export async function recordFailure(ctx, agentKey: string, reason: string) {
         const next = (counters.get(agentKey) ?? 0) + 1;
         counters.set(agentKey, next);
         await ctx.db.execute("INSERT INTO editor_agent_failures (agent_key, reason, consecutive) VALUES ($1, $2, $3)", [agentKey, reason, next]);
         if (next >= MAX_CONSECUTIVE_FAILURES) {
           await ctx.agents.pause(agentKey, `${MAX_CONSECUTIVE_FAILURES} consecutive failures: ${reason}`);
         }
         return next;
       }
       export function recordSuccess(agentKey: string) { counters.set(agentKey, 0); }
       ```
       Note: counter is in-memory per worker process; survives the worker, not host restarts. The `editor_agent_failures` table is the durable audit log. For v1 dogfood this is acceptable; v2 reads the last 3 rows from the DB on worker boot to rebuild state.

    5. **src/worker/agents/compile-tldr.ts**:
       ```typescript
       import crypto from 'node:crypto';
       import { upsertTldr, getTldrByScope } from '../db/tldr-cache.js';
       import { recordFailure, recordSuccess } from './circuit-breaker.js';
       import { EDITOR_WRITE_TAG } from './self-loop-filter.js';

       export const MAX_TOKENS = 4000;  // D-05 placeholder; instrument during dogfood

       function estimateTokens(text: string): number {
         // ~4 chars per token rule-of-thumb. Replace with tiktoken in v2.
         return Math.ceil(text.length / 4);
       }

       export async function compileTldr(ctx, args: { surface: 'issue'|'situation'|'bulletin'; scopeId: string; inputs: { body: string; comments: string[]; refs: string[] }; agentKey: string; agentId: string }) {
         const inputBlob = JSON.stringify(args.inputs);
         const contentHash = crypto.createHash('sha256').update(args.surface + '\0' + args.scopeId + '\0' + inputBlob).digest('hex');

         // EDITOR-03: idempotency.
         const cached = await getTldrByScope(ctx, args.surface, args.scopeId);
         if (cached && cached.content_hash === contentHash) return cached;

         // EDITOR-05: enforce BEFORE LLM call, not after.
         const inputTokens = estimateTokens(inputBlob);
         if (inputTokens > MAX_TOKENS) {
           await recordFailure(ctx, args.agentKey, `input_tokens=${inputTokens} exceeds MAX_TOKENS=${MAX_TOKENS}`);
           throw new Error(`Editor-Agent input exceeds max_tokens cap (${inputTokens} > ${MAX_TOKENS})`);
         }

         // Invoke LLM adapter (host-mediated). For v1 dogfood, the agent's adapter handles model selection.
         // Adapter call shape varies by SDK version — ctx.agents.invoke or ctx.tools.invokeLlm; verify against kitchen-sink.
         let body: string;
         try {
           body = await ctx.llm.complete({ maxTokens: MAX_TOKENS, prompt: buildPrompt(args) });
           recordSuccess(args.agentKey);
         } catch (err) {
           await recordFailure(ctx, args.agentKey, `llm_error: ${(err as Error).message}`);
           throw err;
         }

         // Schema validation of LLM output (EDITOR-05 "output failed schema validation" failure mode).
         if (typeof body !== 'string' || body.length === 0 || body.length > 8000) {
           await recordFailure(ctx, args.agentKey, `output_schema_invalid: len=${body?.length}`);
           throw new Error('Editor-Agent output failed schema validation');
         }

         const tldr = {
           surface: args.surface,
           scope_id: args.scopeId,
           content_hash: contentHash,
           body,
           generated_at: new Date().toISOString(),
           source_revisions: [contentHash],
           compiled_by_agent_id: args.agentId,
           tags: [EDITOR_WRITE_TAG]   // D-04 self-loop filter tag
         };
         await upsertTldr(ctx, tldr);
         return tldr;
       }

       function buildPrompt(args) { /* …issue body + comments + ref summaries; ≤ MAX_TOKENS estimated input. */ return ''; }
       ```

    6. **src/worker/agents/editor.ts**:
       ```typescript
       import { filterSelfLoopEvents, EDITOR_WRITE_TAG } from './self-loop-filter.js';
       import { compileTldr } from './compile-tldr.js';

       export const EDITOR_AGENT_KEY = 'editor-agent';
       export const EDITOR_AGENT_ID_TAG = 'clarity-pack-editor-agent';

       export async function reconcileEditorAgent(ctx, companyId: string) {
         await ctx.agents.managed.reconcile(EDITOR_AGENT_KEY, companyId);
       }

       // Called when the host invokes the agent via heartbeat. Receives a heartbeat context payload.
       export async function handleEditorHeartbeat(ctx, payload: { companyId: string; events: any[]; agentId: string }) {
         const filtered = filterSelfLoopEvents(payload.events, payload.agentId);  // D-04
         // For each issue mentioned in filtered events, recompute TL;DR if its body content_hash changed.
         const issueIds = Array.from(new Set(filtered.filter(e => e.entity_type === 'issue').map(e => e.entity_id)));
         for (const issueId of issueIds) {
           const issue = await ctx.issues.get(issueId);
           const comments = await ctx.issue.comments.read(issueId);
           const refs = extractRefsFromBody(issue.body);
           await compileTldr(ctx, { surface: 'issue', scopeId: issueId, inputs: { body: issue.body, comments: comments.map(c => c.body), refs }, agentKey: EDITOR_AGENT_KEY, agentId: payload.agentId });
         }
       }
       function extractRefsFromBody(body: string): string[] { return [...body.matchAll(/\bBEAAA-\d+\b/g)].map(m => m[0]); }
       ```

       import * as React from 'react';
       import { RefChip } from '../../primitives/ref-chip.js';

       const REF_PATTERN = /\bBEAAA-\d+\b/g;

       export function ProseWithRefChips({ body }: { body: string }) {
         if (!body) return null;
         const nodes: React.ReactNode[] = [];
         let lastIndex = 0;
         let match: RegExpExecArray | null;
         const re = new RegExp(REF_PATTERN.source, 'g');  // fresh regex (lastIndex isolated)
         while ((match = re.exec(body)) !== null) {
           if (match.index > lastIndex) nodes.push(body.slice(lastIndex, match.index));
           nodes.push(<RefChip key={`ref-${match.index}`} refId={match[0]} />);
           lastIndex = match.index + match[0].length;
         }
         if (lastIndex < body.length) nodes.push(body.slice(lastIndex));
         return <div className="clarity-reader-prose">{nodes}</div>;
       }
