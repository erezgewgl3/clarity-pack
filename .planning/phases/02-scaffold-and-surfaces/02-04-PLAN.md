       import * as React from 'react';
       export function useLeaderElection({ channelName }: { channelName: string }): { isLeader: boolean; available: boolean } {
         const available = typeof BroadcastChannel !== 'undefined';
         const [isLeader, setIsLeader] = React.useState<boolean>(!available);  // fallback: every tab is its own leader
         React.useEffect(() => {
           if (!available) {
             console.warn('[clarity-pack] BroadcastChannel unavailable — falling back to per-tab polling (Situation Room may make N parallel fetches).');
             return;
           }
           const channel = new BroadcastChannel(channelName);
           const me = crypto.randomUUID();
           let leaderId: string | null = null;
           const election = () => channel.postMessage({ kind: 'announce', id: me });
           channel.onmessage = (e) => {
             if (e.data.kind === 'announce') {
               if (!leaderId || e.data.id < leaderId) leaderId = e.data.id;   // lowest UUID wins (deterministic)
             }
             if (e.data.kind === 'leader-response') leaderId = e.data.id;
             setIsLeader(leaderId === me);
           };
           // Re-announce periodically so a closed leader's tab can be replaced.
           const interval = setInterval(election, 10_000);
           election();
           return () => { clearInterval(interval); channel.close(); };
         }, [channelName, available]);
         return { isLeader, available };
       }

       ```typescript
       import * as React from 'react';
       import { usePoll, type PollError } from './use-poll.js';
       import { useLeaderElection } from './use-leader-election.js';

       export function usePollWithLeader<T>(opts: { key: string; fetcher: () => Promise<T>; intervalMs: number; pauseOnHidden?: boolean }) {
         const channelName = `clarity-${opts.key}`;
         const { isLeader, available } = useLeaderElection({ channelName });

         // Leader runs the real fetcher; followers receive null (no scheduled fetch).
         const polled = usePoll<T>({
           key: opts.key,
           fetcher: (isLeader || !available) ? opts.fetcher : async () => null as any,
           intervalMs: opts.intervalMs,
           pauseOnHidden: opts.pauseOnHidden,
         });

         // Follower-receives-leader-payload bridge (revision iteration 2 warning #2 acceptance).
         const [followerData, setFollowerData] = React.useState<T | null>(null);
         const [followerError, setFollowerError] = React.useState<PollError | null>(null);
         React.useEffect(() => {
           if (!available || isLeader) return;
           const channel = new BroadcastChannel(channelName);
           channel.onmessage = (e) => {
             if (e.data?.kind === 'leader-data') setFollowerData(e.data.payload as T);
             if (e.data?.kind === 'leader-error') setFollowerError(e.data.error as PollError);
           };
           return () => channel.close();
         }, [channelName, available, isLeader]);

         // Leader broadcasts its latest data + error to followers.
         React.useEffect(() => {
           if (!available || !isLeader) return;
           const channel = new BroadcastChannel(channelName);
           if (polled.data !== null) channel.postMessage({ kind: 'leader-data', payload: polled.data });
           if (polled.error !== null) channel.postMessage({ kind: 'leader-error', error: polled.error });
           return () => channel.close();
         }, [polled.data, polled.error, channelName, available, isLeader]);

         return {
           data: isLeader || !available ? polled.data : followerData,
           error: isLeader || !available ? polled.error : followerError,
           stale: polled.stale,
           isLeader,
         };
       }
       ```
       Note: this implementation requires that re-broadcasting is idempotent on the follower side (setState with the same object is a no-op for React if the reference doesn't change between effects; consider memoizing by content-hash if real-world thrashing emerges — Phase 3/4 may revisit).
