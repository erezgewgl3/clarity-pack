       import type { BlockerChainResult, Terminal } from '../../shared/types.ts';

       const UUID_RE = /[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}/g;

       export function isUuidShaped(s: string): boolean {
         if (!s || typeof s !== 'string') return false;
         // Reset regex state by using a fresh test
         return /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/.test(s);
       }

       export type IdLookupEntry = { label: string; kind: 'agent' | 'user' };
       export type IdLookup = Map<string, IdLookupEntry>;

       /**
        * Title-case for agent roles. Short all-lowercase strings (2-4 chars)
        * become all-uppercase (acronyms like 'ceo' → 'CEO'). Longer strings get
        * first-letter-uppercase only ('editor' → 'Editor'). Mixed/cap input is
        * passed through unchanged.
        */
       function titleCaseRole(raw: string): string {
         if (!raw) return raw;
         if (raw !== raw.toLowerCase()) return raw;
         if (raw.length >= 2 && raw.length <= 4) return raw.toUpperCase();
         return raw[0].toUpperCase() + raw.slice(1);
       }

       export function buildIdLookup({ agents, users }: {
         agents: Array<{ user_id?: string; id?: string; role?: string }>;
         users: Array<{ id: string; name: string }>;
       }): IdLookup {
         const out: IdLookup = new Map();
         for (const a of agents) {
           const id = a.user_id ?? a.id;
           if (!id) continue;
           out.set(id, { label: titleCaseRole(a.role ?? 'agent'), kind: 'agent' });
         }
         for (const u of users) {
           if (!u.id) continue;
           out.set(u.id, { label: u.name ?? u.id.slice(0, 8), kind: 'user' });
         }
         return out;
       }

       /**
        * Rewrites terminal.label so it contains no raw UUIDs. Three passes:
        * (1) For HUMAN_ACTION_ON with userId === '__unowned__': rewrite to
        *     "<resolved-or-agent> has no owner assigned".
        * (2) For any UUID substring inside the label: replace with lookup label
        *     or short-form 'agent#abcd1234' if not in lookup.
        * (3) Preserve other terminal kinds untouched except UUID substring
        *     substitution (rare but defensive).
        *
        * Pure function — returns a NEW chain + NEW terminal; never mutates.
        */
       export function humanizeChain(chain: BlockerChainResult, lookup: IdLookup): BlockerChainResult {
         const t = chain.terminal;
         let newLabel = t.label;

         // Step 1: __unowned__ → "no owner assigned" form
         if (t.kind === 'HUMAN_ACTION_ON' && t.userId === '__unowned__') {
           // Try to extract the first UUID in the existing label as the agent it refers to.
           const m = t.label.match(/[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}/);
           const agentLabel = m && lookup.get(m[0])?.label;
           if (agentLabel) {
             newLabel = `${agentLabel} has no owner assigned`;
           } else {
             newLabel = 'Agent has no owner assigned';
           }
         } else {
           // Step 2: substitute every UUID with its lookup label, or short form
           newLabel = t.label.replace(UUID_RE, (uuid) => {
             const entry = lookup.get(uuid);
             if (entry) return entry.label;
             return `agent#${uuid.slice(0, 8)}`;
           });
         }

         // Belt-and-suspenders: if any UUID slipped through, replace with short form
         newLabel = newLabel.replace(UUID_RE, (uuid) => `agent#${uuid.slice(0, 8)}`);

         const newTerminal: Terminal = { ...t, label: newLabel };
         return { ...chain, terminal: newTerminal };
       }
