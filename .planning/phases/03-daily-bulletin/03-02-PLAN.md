       // src/worker/bulletin/facts-table.ts
       // Plan 03-02 — Pure facts-table extraction + slot interpolation.
       //
       // CONTEXT.md D-14: structured slots, not prose-extracted numbers. The
       // LLM prompt receives the factsTable as DATA, never as instruction; the
       // prose it emits uses `{{NUMBER:key}}` placeholders that pure code
       // replaces — the LLM never types a number into the prose itself.

       import type { FactsTable } from '../../shared/types.ts';

       export type FactsInput = {
         rows: Record<string, number | string>;  // Input data — keyed by slot name
         // Optional: slot definitions, for richer formats
         slotDefs?: Record<string, { sql: string; params: unknown[]; format: 'currency' | 'count' | 'pct' | 'ratio' }>;
       };

       /**
        * Pure function. Maps raw SQL-result rows into a FactsTable shape that
        * the pass-1 LLM prompt + pass-2 verifier can both consume.
        */
       export function computeFactsTable(input: FactsInput): FactsTable {
         const out: FactsTable = {};
         for (const [key, value] of Object.entries(input.rows)) {
           const def = input.slotDefs?.[key];
           out[key] = {
             sql: def?.sql ?? '',
             params: def?.params ?? [],
             value: value as number,
             format: def?.format ?? (typeof value === 'string' ? 'count' : 'count'),
           };
         }
         return out;
       }

       /**
        * Pure function. Replaces every `{{NUMBER:key}}` placeholder in `prose`
        * with the formatted value from `facts[key]`. Format-aware: 'currency'
        * → en-US currency, 'pct' → one-decimal percent, 'count' → raw integer,
        * 'ratio' → "N/M".
        *
        * If a placeholder references a key not in `facts`, throws an Error
        * tagged with the unknown slot key — the verifier downstream catches
        * and converts to `{ok:false, kind:'UNKNOWN_SLOT'}`.
        */
       export function replaceSlots(prose: string, facts: FactsTable): string {
         return prose.replace(/\{\{NUMBER:([a-zA-Z0-9_]+)\}\}/g, (_, key: string) => {
           const fact = facts[key];
           if (!fact) {
             const err = new Error(`UNKNOWN_SLOT:${key}`);
             (err as Error & { slot?: string }).slot = key;
             throw err;
           }
           switch (fact.format) {
             case 'currency':
               return new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD', maximumFractionDigits: 0 }).format(Number(fact.value));
             case 'pct':
               return `${(Number(fact.value) * 100).toFixed(1)}%`;
             case 'count':
               return new Intl.NumberFormat('en-US', { maximumFractionDigits: 0 }).format(Number(fact.value));
             case 'ratio':
               return String(fact.value);
             default:
               return String(fact.value);
           }
         });
       }

       // Plan 03-02 — Pass-2 deterministic verifier (CONTEXT.md D-15).
       //
       // NEVER an LLM. Pure function: re-runs every standing_numbers[i].sql
       // and compares actual to claimed. Returns typed VerifierResult. Three
       // consecutive ok:false at the CALLER layer (compile-bulletin.ts)
       // trips the existing circuit-breaker via recordFailure(agentKey='bulletin-compile').

       import { STANDING_NUMBER_SLOTS } from './standing-numbers.ts';

       export type SqlClient = {
         query<T>(sql: string, params?: unknown[]): Promise<T[]>;
       };

       const PCT_TOLERANCE = 0.01;     // ±1 percentage point absolute
       const RATIO_TOLERANCE = 0.01;

       /**
        * Pure-async function. Iterates draft.standingNumbers; for each slot,
        * looks up the canonical SQL in STANDING_NUMBER_SLOTS by key; re-runs
        * via sqlClient; compares actual to slot.value (the LLM's claim, but
        * since the LLM only sees factsTable values, in practice the LLM
        * never types these — the slot.value comes from the factsTable
        * directly via the pass-1 structured emission).
        *
        * Verifier-side this distinction doesn't matter — we always re-run.
        */
       export async function verifyDraft(
         draft: BulletinDraft,
         sqlClient: SqlClient,
         companyId: string,
       ): Promise<VerifierResult> {
         // (a) Schema-level: every prose `{{NUMBER:X}}` must map to a slot
         // key. The actual placeholder scan happens in pass-1's replaceSlots
         // (which throws on UNKNOWN_SLOT and the caller catches into
         // VerifierResult). The verifier itself only re-checks numeric slots.

         const mismatches: Array<{ slot: string; claimed: unknown; actual: unknown; tolerance: number }> = [];

         for (const claimed of draft.standingNumbers) {
           const slotDef = STANDING_NUMBER_SLOTS.find((s) => s.key === claimed.key);
           if (!slotDef) {
             return { ok: false, kind: 'UNKNOWN_SLOT', slot: claimed.key };
           }
           const params = slotDef.params.map((p) => (p === '<companyId>' ? companyId : p));
           let actual: number;
           try {
             const rows = await sqlClient.query<{ value: number }>(slotDef.sql, params);
             actual = Number(rows[0]?.value ?? 0);
           } catch {
             // Treat as mismatch with tolerance 0
             mismatches.push({ slot: claimed.key, claimed: claimed.value, actual: 'query_failed', tolerance: 0 });
             continue;
           }

           const tolerance = claimed.format === 'pct' ? PCT_TOLERANCE : claimed.format === 'ratio' ? RATIO_TOLERANCE : 0;
           const diff = Math.abs(Number(claimed.value) - actual);
           if (diff > tolerance) {
             mismatches.push({ slot: claimed.key, claimed: claimed.value, actual, tolerance });
           }
         }

         if (mismatches.length === 0) return { ok: true };
         return { ok: false, mismatches };
       }
       ```

        * Throws if `body` is not a well-formed BulletinDraft. Asserts every
        * required key + every prose `{{NUMBER:X}}` resolves through replaceSlots
        * (which throws on UNKNOWN_SLOT).
        */
       export function validateDraftSchema(body: unknown, facts: FactsTable): asserts body is BulletinDraft {
         if (!body || typeof body !== 'object') {
           throw new Error('BulletinDraft failed schema validation (not an object)');
         }
         const d = body as Record<string, unknown>;
         if (!d.masthead || typeof d.masthead !== 'object') throw new Error('BulletinDraft.masthead missing');
         if (!Array.isArray(d.actionInbox)) throw new Error('BulletinDraft.actionInbox must be array');
         if (!Array.isArray(d.departments)) throw new Error('BulletinDraft.departments must be array');
         if (!Array.isArray(d.standingNumbers)) throw new Error('BulletinDraft.standingNumbers must be array');
         if (!Array.isArray(d.lineageThreads)) throw new Error('BulletinDraft.lineageThreads must be array');

         // Scan all department prose for `{{NUMBER:X}}` placeholders; throw UNKNOWN_SLOT if any reference an unknown key.
         for (const dept of d.departments as Array<{ editorialSummary?: string }>) {
           if (typeof dept.editorialSummary === 'string') {
             try {
               replaceSlots(dept.editorialSummary, facts);
             } catch (e) {
               throw e;   // already tagged with .slot
             }
           }
         }
       }

       export async function compilePass1(
         ctx: CompilePass1Ctx,
         args: CompilePass1Args,
       ): Promise<BulletinDraft> {
         // Build prompt: factsTable as JSON data + structural instructions.
         const prompt = buildPrompt(args);
         const inputTokens = estimateTokens(prompt);
         if (inputTokens > MAX_BULLETIN_TOKENS) {
           await recordFailure(ctx, {
             agentKey: BULLETIN_COMPILE_AGENT_KEY,
             agentId: EDITOR_AGENT_ID_TAG,
             companyId: args.companyId,
             reason: `input_tokens=${inputTokens} exceeds MAX_BULLETIN_TOKENS=${MAX_BULLETIN_TOKENS}`,
           });
           throw new Error(`Bulletin compile input exceeds max_tokens cap (${inputTokens} > ${MAX_BULLETIN_TOKENS})`);
         }

         const llm = args.llm ?? ctx.llm;
         if (!llm) {
           throw new Error('compilePass1 called without an LLM adapter');
         }

         let raw: string;
         try {
           raw = await llm.complete({ maxTokens: MAX_BULLETIN_TOKENS, prompt });
         } catch (err) {
           await recordFailure(ctx, {
             agentKey: BULLETIN_COMPILE_AGENT_KEY,
             agentId: EDITOR_AGENT_ID_TAG,
             companyId: args.companyId,
             reason: `LLM call threw: ${(err as Error).message}`,
           });
           throw err;
         }

         let parsed: unknown;
         try {
           parsed = JSON.parse(raw);
         } catch (err) {
           await recordFailure(ctx, {
             agentKey: BULLETIN_COMPILE_AGENT_KEY,
             agentId: EDITOR_AGENT_ID_TAG,
             companyId: args.companyId,
             reason: `LLM output not valid JSON: ${(err as Error).message}`,
           });
           throw new Error('compilePass1: LLM output not valid JSON');
         }

         try {
           validateDraftSchema(parsed, args.factsTable);
         } catch (err) {
           await recordFailure(ctx, {
             agentKey: BULLETIN_COMPILE_AGENT_KEY,
             agentId: EDITOR_AGENT_ID_TAG,
             companyId: args.companyId,
             reason: `Draft schema invalid: ${(err as Error).message}`,
           });
           throw err;
         }

         await recordSuccess(ctx, {
           agentKey: BULLETIN_COMPILE_AGENT_KEY,
           companyId: args.companyId,
         });

         return parsed as BulletinDraft;
       }

       function buildPrompt(args: CompilePass1Args): string {

       // src/shared/bulletin-rendering.ts
       // Plan 03-02 — Pure markdown rendering of a BulletinDraft into the
       // canonical body stored in public.issues.description per D-16.

       import type { BulletinDraft, StandingNumberRow, LineageThread, ActionInboxCard } from './types.ts';

       export function renderBulletinIssueBody(draft: BulletinDraft): string {
         const parts: string[] = [];

         // ----- Masthead -----
         const m = draft.masthead;
         parts.push(`# The Bulletin · Vol. ${m.volume} · No. ${m.number}`);
         parts.push('');
         parts.push(`*${m.weekday}, ${m.dateText} · 06:30 ET · prepared for ${m.prepareForName}, Editor-in-Chief · Operations Cycle ${m.cycleNumber} · Auto-compiled*`);
         parts.push('');

         // ----- Action Inbox -----
         if (draft.actionInbox.length > 0) {
           parts.push('## Requires Your Decision');
           parts.push('');
           for (const card of draft.actionInbox) {
             parts.push(`### ${card.title}`);
             parts.push(`*${card.department} · ${card.ageText}*`);
             parts.push('');
             parts.push(card.summary);
             parts.push('');
             parts.push(`Actions: Approve · Open · Decline`);
             parts.push('');
           }
         }

         // ----- Department sections -----
         parts.push(`## Yesterday's Operations`);
         parts.push('');
         for (const dept of draft.departments) {
           parts.push(`### ${dept.name}`);
           parts.push('');
           if (dept.items.length === 0) {
             parts.push('*· no items ·*');
             parts.push('');
             continue;
           }
           if (dept.editorialSummary) {
             parts.push(dept.editorialSummary);
             parts.push('');
           }
           for (const item of dept.items) {
             parts.push(`- **${item.title}** — ${item.timeText}`);
             parts.push(`  - ${item.bylineHtml.replace(/<[^>]+>/g, '')}`);   // strip any HTML tags (we emit markdown only)
             if (item.lineageInline) parts.push(`  - Lineage: ${item.lineageInline}`);
             if (item.note) parts.push(`  - ${item.note}`);
           }
           parts.push('');
         }

         // ----- Standing Numbers -----
         parts.push('## Standing Numbers');
         parts.push('');
         for (const sn of draft.standingNumbers) {
           parts.push(`- **${sn.displayName}**: ${formatStandingValue(sn)}`);
         }
         parts.push('');

         // ----- Lineage Threads -----
         if (draft.lineageThreads.length > 0) {
           const featured = draft.lineageThreads[0];
           parts.push(`## One artifact, end-to-end — ${featured.id}`);
           parts.push('');
           const arrows = featured.nodes.map((n, i) => `${i + 1}. **${n.name}** (${n.time}) — ${n.detail}${n.isTerminal ? ' [TERMINAL]' : ''}`);
           parts.push(...arrows);
           if (featured.truncatedCount > 0) parts.push(`*…and ${featured.truncatedCount} more steps*`);
           parts.push('');
         }

         return parts.join('\n');
       }

       function formatStandingValue(sn: StandingNumberRow): string {
         switch (sn.format) {
           case 'currency':
             return new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD', maximumFractionDigits: 0 }).format(sn.value);
           case 'pct':
             return `${(sn.value * 100).toFixed(1)}%`;
           case 'count':
             return new Intl.NumberFormat('en-US').format(sn.value);
           case 'ratio':
             return String(sn.value);
           default:
             return String(sn.value);
         }
       }
       ```

       // src/worker/bulletin/publish.ts
       // Plan 03-02 — Two-phase publish of a verified BulletinDraft (BULL-09).
       //
       // Order:
       // 1. INSERT bulletins (compile_status='attempting', content_hash, next_due_at,
       //    draft_json = the verified BulletinDraft per W3/W4)
       //    — UNIQUE(next_due_at, content_hash) enforces idempotency
       // 2. ctx.issues.create({title, description, tags=[clarity:bulletin, clarity:bulletin-issue, cycle:N], authorAgentId})
       // 3. UPDATE bulletins SET published_issue_id, published_at, compile_status='published'
       //
       // On any failure: leave bulletins.compile_status='attempting' for the
       // retry loop to clean up. No partial-published rows.

       import crypto from 'node:crypto';

       import type { PluginIssuesClient, PluginDatabaseClient } from '@paperclipai/plugin-sdk';
       import { formatInTimeZone } from 'date-fns-tz';
       import { BULLETIN_TZ } from './next-due-at.ts';
       import { renderBulletinIssueBody } from '../../shared/bulletin-rendering.ts';
       import type { BulletinDraft } from '../../shared/types.ts';

       export type PublishBulletinCtx = {
         db: PluginDatabaseClient;
         issues: PluginIssuesClient;
         logger?: { info?(...a: unknown[]): void; warn?(...a: unknown[]): void };
       };

       export type PublishBulletinArgs = {
         companyId: string;
         cycleNumber: number;
         nextDueAtIso: string;
         editorAgentId: string;
         draft: BulletinDraft;
         compiledAt: Date;
       };

       export type PublishResult =
         | { kind: 'published'; cycleNumber: number; publishedIssueId: string; publishedAt: string }
         | { kind: 'duplicate'; cycleNumber: number }
         | { kind: 'failed'; reason: string };

       function syncHash(s: string): string {
         return crypto.createHash('sha256').update(s).digest('hex').slice(0, 16);
       }

       export async function publishBulletin(
         ctx: PublishBulletinCtx,
         args: PublishBulletinArgs,
       ): Promise<PublishResult> {
         const body = renderBulletinIssueBody(args.draft);
         const contentHash = syncHash(body);
         const weekday = formatInTimeZone(args.compiledAt, BULLETIN_TZ, 'EEEE');
         const dateText = formatInTimeZone(args.compiledAt, BULLETIN_TZ, 'yyyy-MM-dd');
         const title = `Bulletin No. ${args.cycleNumber} — ${weekday}, ${dateText}`;

         // ---- Phase 1: INSERT attempting (idempotency check) ----
         try {
           await ctx.db.execute(
             `INSERT INTO plugin_clarity_pack_cdd6bda4bd.bulletins
               (cycle_number, company_id, next_due_at, compiled_at, compile_status, content_hash, lineage_thread_json, draft_json)
              VALUES ($1, $2, $3, $4, 'attempting', $5, $6::jsonb, $7::jsonb)
              ON CONFLICT (next_due_at, content_hash) DO NOTHING`,
             [args.cycleNumber, args.companyId, args.nextDueAtIso, args.compiledAt.toISOString(), contentHash, JSON.stringify(args.draft.lineageThreads), JSON.stringify(args.draft)],
           );  // W3/W4: draft_json stores the full verified BulletinDraft so Plan 03-03's bulletin-by-cycle handler returns typed props with NO markdown re-parser
         } catch (e) {
           ctx.logger?.warn?.('publishBulletin: phase 1 INSERT failed', { err: (e as Error).message });
           return { kind: 'failed', reason: `bulletins INSERT failed: ${(e as Error).message}` };
         }

         // Confirm we own this row (idempotency: if another concurrent fire already inserted, our INSERT was a no-op)
         const ownsRow = await ctx.db.query<{ compile_status: string }>(
           'SELECT compile_status FROM plugin_clarity_pack_cdd6bda4bd.bulletins WHERE next_due_at = $1 AND content_hash = $2',
           [args.nextDueAtIso, contentHash],
         );
         if (ownsRow[0]?.compile_status === 'published') {
           return { kind: 'duplicate', cycleNumber: args.cycleNumber };
         }

         // ---- Phase 2: ctx.issues.create ----
         let issue: { id: string } | null;
         try {
           issue = await ctx.issues.create({
             companyId: args.companyId,
             title,
             description: body,
             tags: ['clarity:bulletin', 'clarity:bulletin-issue', `cycle:${args.cycleNumber}`],
             authorAgentId: args.editorAgentId,
           } as Parameters<PluginIssuesClient['create']>[0]);
         } catch (e) {
           ctx.logger?.warn?.('publishBulletin: ctx.issues.create threw', { err: (e as Error).message });
           // Leave bulletins.compile_status='attempting' for retry loop.
           return { kind: 'failed', reason: `ctx.issues.create threw: ${(e as Error).message}` };
         }

         if (!issue || !issue.id) {
           return { kind: 'failed', reason: 'ctx.issues.create returned null' };
         }

         // ---- Phase 3: UPDATE bulletins SET compile_status='published' ----
         const publishedAtIso = new Date().toISOString();
         try {
           await ctx.db.execute(
             `UPDATE plugin_clarity_pack_cdd6bda4bd.bulletins
                SET published_issue_id = $1, published_at = $2, compile_status = 'published', verified_at = $2
              WHERE next_due_at = $3 AND content_hash = $4`,
             [issue.id, publishedAtIso, args.nextDueAtIso, contentHash],
           );
         } catch (e) {
           ctx.logger?.warn?.('publishBulletin: phase 3 UPDATE failed', { err: (e as Error).message });
           // Issue is created but metadata didn't flip; next cycle's compile will see the orphan and resolve.
           return { kind: 'failed', reason: `UPDATE published failed: ${(e as Error).message}` };
         }

         return { kind: 'published', cycleNumber: args.cycleNumber, publishedIssueId: issue.id, publishedAt: publishedAtIso };
       }
