       ```typescript
       // src/worker/bulletin/lineage-grouper.ts
       // Plan 03-03 — Temporal+actor proximity heuristic for lineage threads
       // (D-21 fallback since SDK has no caused_by_activity_id field).
       //
       // Pure deterministic code. Same input → byte-equal output. No LLM.

       import type { LineageThread } from '../../shared/types.ts';

       export type ActivityEvent = {
         id: string;
         entityId: string;       // issueId, agentId, document-id, etc.
         actorId: string;         // agent or user id
         actorChain?: string;    // optional "parent → child" thread marker; default = actorId
         timestamp: string;       // ISO
         message: string;
         name?: string;           // optional display name
         detail?: string;
       };

       const DEFAULT_MAX_DELTA_SEC = 300;
       const NODE_TRUNCATE_AT = 8;

       /**
        * Pure function. Groups activities by (entityId, actorChain) into clusters
        * where consecutive events within the same cluster are at most
        * `opts.maxDeltaSec` apart. Returns deterministic LineageThread[].
        */
       export function groupLineageThreads(
         activities: ActivityEvent[],
         opts: { maxDeltaSec?: number } = {},
       ): LineageThread[] {
         if (!Array.isArray(activities) || activities.length === 0) return [];
         const maxDeltaMs = (opts.maxDeltaSec ?? DEFAULT_MAX_DELTA_SEC) * 1000;

         // Deterministic sort: by entityId, then by (actorChain ?? actorId), then by timestamp.
         const sorted = [...activities].sort((a, b) => {
           if (a.entityId !== b.entityId) return a.entityId < b.entityId ? -1 : 1;
           const aChain = a.actorChain ?? a.actorId;
           const bChain = b.actorChain ?? b.actorId;
           if (aChain !== bChain) return aChain < bChain ? -1 : 1;
           return a.timestamp < b.timestamp ? -1 : a.timestamp > b.timestamp ? 1 : 0;
         });

         const threads: LineageThread[] = [];
         let currentGroup: ActivityEvent[] = [];
         let currentKey: string | null = null;

         function flush() {
           if (currentGroup.length === 0) return;
           const head = currentGroup[0];
           const nodes = currentGroup.slice(0, NODE_TRUNCATE_AT).map((ev, i) => ({
             time: ev.timestamp.slice(11, 16),   // HH:MM
             name: ev.name ?? ev.actorId,
             detail: ev.detail ?? ev.message,
             isTerminal: i === Math.min(currentGroup.length, NODE_TRUNCATE_AT) - 1,
           }));
           const truncatedCount = Math.max(0, currentGroup.length - NODE_TRUNCATE_AT);
           threads.push({
             id: `${head.entityId}:${head.actorChain ?? head.actorId}:${head.timestamp}`,
             entityId: head.entityId,
             nodes,
             truncatedCount,
           });
           currentGroup = [];
         }

         for (const ev of sorted) {
           const key = `${ev.entityId}::${ev.actorChain ?? ev.actorId}`;
           if (key !== currentKey) {
             flush();
             currentKey = key;
             currentGroup.push(ev);
             continue;
           }
           const last = currentGroup[currentGroup.length - 1];
           const delta = new Date(ev.timestamp).getTime() - new Date(last.timestamp).getTime();
           if (delta > maxDeltaMs) {
             flush();
             currentGroup.push(ev);
           } else {
             currentGroup.push(ev);
           }
         }
         flush();

         return threads;
       }
       ```

       import type { ActionInboxCard } from '../../shared/types.ts';

       const ACTION_INBOX_WINDOW_DAYS = 30;
       const ACTION_INBOX_MAX = 50;

       export type ActionInboxCtx = {
         issues: PluginIssuesClient;
         db: PluginDatabaseClient;
         logger?: { warn?(...a: unknown[]): void };
       };

       export type ActionInboxArgs = {
         companyId: string;
         viewerUserId: string;
         now?: Date;
       };

       export async function queryActionInbox(
         ctx: ActionInboxCtx,
         args: ActionInboxArgs,
       ): Promise<ActionInboxCard[]> {
         const now = args.now ?? new Date();
         let issues: Awaited<ReturnType<PluginIssuesClient['list']>>;
         try {
           issues = await ctx.issues.list({ companyId: args.companyId, status: 'blocked' as never });
         } catch (e) {
           ctx.logger?.warn?.('action-inbox: ctx.issues.list failed', { err: (e as Error).message });
           return [];
         }

         const cutoff = now.getTime() - ACTION_INBOX_WINDOW_DAYS * 24 * 60 * 60 * 1000;

         // Pre-fetch department membership once per call.
         let memberships: Array<{ employee_user_id: string; department: string }>;
         try {
           memberships = await ctx.db.query(
             'SELECT employee_user_id, department FROM plugin_clarity_pack_cdd6bda4bd.clarity_department_membership WHERE company_id = $1',
             [args.companyId],
           );
         } catch {
           memberships = [];
         }
         const deptMap = new Map(memberships.map((m) => [m.employee_user_id, m.department]));

         const out: ActionInboxCard[] = [];
         for (const i of issues) {
           if (!i.assigneeUserId || i.assigneeUserId !== args.viewerUserId) continue;
           const state = i.blockerAttention?.state;
           if (state !== 'needs_attention' && state !== 'stalled') continue;
           if (new Date(i.updatedAt).getTime() < cutoff) continue;

           const ageMs = now.getTime() - new Date(i.awaitingSince ?? i.updatedAt).getTime();
           out.push({
             issueId: i.id,
             identifier: i.identifier,
             title: i.title,
             department: deptMap.get(i.assigneeUserId) ?? 'Builder',
             ageMs,
             ageText: formatAge(ageMs),
             summary: truncate(i.description, 280),
           });
           if (out.length >= ACTION_INBOX_MAX) break;
         }
         return out;
       }

       function formatAge(ms: number): string {
         const mins = Math.floor(ms / 60000);
         if (mins < 60) return `${mins}m`;
         const hours = Math.floor(mins / 60);
         if (hours < 24) return `${hours}h`;
         const days = Math.floor(hours / 24);
         return `${days}d`;
       }

       function truncate(s: string | undefined | null, n: number): string {
         if (!s) return '';
         return s.length > n ? `${s.slice(0, n - 1)}…` : s;
       }

                   companyId={props.companyId}
                   userId={props.userId}

       function ActionInboxCardView({ card, companyId, userId, onActionComplete }: { card: ActionInboxCard; companyId: string; userId: string; onActionComplete?: () => void }): React.ReactElement {
         const approve = usePluginAction('bulletin.action.approve');
         const decline = usePluginAction('bulletin.action.decline');
         const nav = useHostNavigation();
         const issueUrl = `/issues/${card.identifier}`;

         const handleApprove = async () => {
           await approve.run({ issueId: card.issueId, companyId, userId });
           onActionComplete?.();
         };
         const handleDecline = async () => {
           await decline.run({ issueId: card.issueId, companyId, userId });
           onActionComplete?.();
         };

         return (
           <article className="clarity-bulletin-action-card">
             <div className="clarity-bulletin-action-meta">
               <span className="clarity-bulletin-dept-tag">{card.department}</span>
               <span className="clarity-bulletin-action-time">{card.ageText}</span>
             </div>
             <h3 className="clarity-bulletin-action-card-h3">{card.title}</h3>
             <p className="clarity-bulletin-summary">{card.summary}</p>
             <div className="clarity-bulletin-actions">
               <button type="button" className="clarity-bulletin-btn" onClick={handleApprove} disabled={approve.loading}>
                 Approve
               </button>
               <a className="clarity-bulletin-btn clarity-bulletin-btn-secondary" {...nav.linkProps(issueUrl)}>

                 Decline
               </button>
             </div>
           </article>
         );
       }

       }
