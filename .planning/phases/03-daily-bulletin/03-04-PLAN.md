        ```javascript
        #!/usr/bin/env node
        // Plan 03-04 — Coexistence Check 07: disabling clarity-pack leaves
        // every `Bulletin No. N` issue searchable in classic Paperclip;
        // bulletins metadata table is NOT dropped on uninstall.

        import { readFileSync, readdirSync } from 'node:fs';
        import path from 'node:path';
        import { fileURLToPath } from 'node:url';

        const HERE = path.dirname(fileURLToPath(import.meta.url));
        const REPO_ROOT = path.resolve(HERE, '..', '..');

        function fail(msg) {
          console.error(`FAIL: ${msg}`);
          process.exit(1);
        }

        // Check 1: No migration drops or alters the bulletins / bulletin_errata / public.issues schema.
        const migrationsDir = path.join(REPO_ROOT, 'migrations');
        const sqlFiles = readdirSync(migrationsDir).filter((f) => f.endsWith('.sql'));
        for (const f of sqlFiles) {
          const sql = readFileSync(path.join(migrationsDir, f), 'utf8');
          if (/\bDROP\s+TABLE\b/i.test(sql)) {
            fail(`migration ${f} contains DROP TABLE — violates additive-only invariant (COEXIST-03 + BULL-09)`);
          }
          if (/DELETE\s+FROM\s+public\.issues/i.test(sql)) {
            fail(`migration ${f} contains DELETE FROM public.issues — would destroy bulletin issues on disable`);
          }
          if (/\bDROP\s+SCHEMA\s+plugin_clarity_pack/i.test(sql)) {
            fail(`migration ${f} attempts DROP SCHEMA — host validator should reject, but defense-in-depth`);
          }
        }

        // Check 2: No code path in the plugin source calls ctx.issues.delete on a bulletin issue.
        // (Plugin only ever writes via ctx.issues.create + ctx.issues.update + ctx.issues.createComment.)
        const srcWorker = path.join(REPO_ROOT, 'src', 'worker');
        const walkDirs = (dir) => {
          const out = [];
          for (const ent of readdirSync(dir, { withFileTypes: true })) {
            const full = path.join(dir, ent.name);
            if (ent.isDirectory()) out.push(...walkDirs(full));
            else if (ent.name.endsWith('.ts') || ent.name.endsWith('.tsx') || ent.name.endsWith('.mjs')) out.push(full);
          }
          return out;
        };
        const workerFiles = walkDirs(srcWorker);
        for (const f of workerFiles) {
          const src = readFileSync(f, 'utf8');
          if (/ctx\.issues\.delete\(/.test(src)) {
            fail(`${f} calls ctx.issues.delete — bulletin issues must survive plugin disable (BULL-09 + COEXIST-03)`);
          }
        }

        // Check 3: The manifest does NOT declare a destructive `onUninstall` hook.
        // (PLUGIN_SPEC §21 — uninstall hooks are out of scope per CONTEXT.md D-16.)
        const manifestPath = path.join(REPO_ROOT, 'src', 'manifest.ts');
        const manifest = readFileSync(manifestPath, 'utf8');
        if (/onUninstall|destructiveUninstall|purgeOnDisable/i.test(manifest)) {
          fail('src/manifest.ts declares an uninstall hook — bulletin data must survive disable');
        }

        console.log('PASS: 07-bulletin-disable — clarity-pack uninstall preserves all prior bulletin issues + metadata');
        process.exit(0);
        ```
