**Task A — migrations/0006_chat.sql (RED first: write test/migrations/0006-chat-schema.test.mjs, then the migration).**
Copy the `0004_bulletin.sql` ll. 1-42 comment header verbatim, change the plan reference to "Plan 04-02 — Employee Chat data layer" and the table list. Then the two tables exactly per RESEARCH.md Pattern 1 (ll. 218-245):

```sql
CREATE TABLE IF NOT EXISTS plugin_clarity_pack_cdd6bda4bd.chat_topics (
  topic_id          text PRIMARY KEY,
  company_id        text NOT NULL,
  issue_id          text NOT NULL,
  parent_issue_id   text NOT NULL,
  employee_agent_id text NOT NULL,
  title             text NOT NULL,
  last_activity_at  timestamptz NOT NULL DEFAULT now(),
  archived          boolean NOT NULL DEFAULT false,
  created_at        timestamptz NOT NULL DEFAULT now(),
  UNIQUE (company_id, issue_id)
);

CREATE TABLE IF NOT EXISTS plugin_clarity_pack_cdd6bda4bd.chat_messages (
  message_uuid      text PRIMARY KEY,
  company_id        text NOT NULL,
  topic_issue_id    text NOT NULL,
  comment_id        text,
  sender_kind       text NOT NULL CHECK (sender_kind IN ('user','agent')),
  supersedes_uuid   text,
  pinned            boolean NOT NULL DEFAULT false,
  sent_at           timestamptz NOT NULL DEFAULT now()
);
```
