```typescript
import type { PluginIssuesClient, PluginLogger } from '@paperclipai/plugin-sdk';

export type TrueTaskCtx = {
  issues: PluginIssuesClient;
  logger?: PluginLogger;
};

export async function createTrueTask(
  ctx: TrueTaskCtx,
  input: {
    companyId: string;
    title: string;
    description: string;
    assigneeAgentId: string;        // D-06 — NEVER omitted
    topicIssueId: string;           // for D-07 back-link + marker comment
    sourceCommentId: string | null; // chat message the task came from
    employeeName: string;           // marker-comment copy
  },
): Promise<{ issueId: string }> {
  // D-05 — TOP-LEVEL: NO parentId. Anti-pattern from chat-promote.ts.
  const issue = await ctx.issues.create({
    companyId: input.companyId,
    title: input.title,
    description: input.description,
    status: 'todo',
    assigneeAgentId: input.assigneeAgentId,
    originKind: 'plugin:clarity-pack',
    originId: `chat-task:${input.topicIssueId}:${input.sourceCommentId ?? 'composer'}`,
  });
  // D-07 marker comment on the topic — first-class conversational content.
  await ctx.issues.createComment(
    input.topicIssueId,
    `Task created — ${issue.id}, assigned to ${input.employeeName}.`,
    input.companyId,
  );
  return { issueId: issue.id };
}
```

```typescript
// src/worker/chat/comment-classify.ts
//
// D-14/D-15 — host-field-first comment classification. Pure function;
// unit-testable in node --test. The IssueComment shape comes from
// packages/shared/src/types/issue.ts (paperclipai/paperclip@master) —
// `authorType: 'user' | 'agent' | 'system'` (constants.ts) +
// `presentation.kind: 'message' | 'system_notice'` (constants.ts).
// RECOVERY service authors disposition / recovery-owner / finish_
// successful_run_handoff notices with authorType: 'system' (verified
// in server/src/services/recovery/service.ts).

export type CommentClass = 'conversation' | 'runtime-noise';

type CommentLike = {
  authorType?: string | null;
  presentation?: { kind?: string | null } | null;
  body?: string;
};

export function classifyComment(c: CommentLike): CommentClass {
  // PRIMARY (HIGH confidence) — host discriminator.
  if (c.authorType === 'system') return 'runtime-noise';
  if (c.presentation?.kind === 'system_notice') return 'runtime-noise';
  // FALLBACK (defense-in-depth) — narrow body-pattern blocklist.
  const RUNTIME_PHRASES = [
    'needs a disposition',
    'blocked on a recovery owner',
    'finish_successful_run_handoff',
    'exhausted the bounded corrective handoff',
  ];
  const body = (c.body ?? '').toLowerCase();
  if (RUNTIME_PHRASES.some((p) => body.includes(p.toLowerCase()))) {
    return 'runtime-noise';
  }
  return 'conversation';
}
```

### `src/worker/chat/topic-watchdog.ts` (NEW — service helper)

**Analog:** the `chat-send.ts:92-110` auto-reopen block (the exact same `ctx.issues.get` → status-check → `ctx.issues.update` pattern, just extended with the `requestWakeup` nudge from research).

**Pattern (RESEARCH §Pattern 3 — copy verbatim, adapt the status flip target):**
```typescript
// src/worker/chat/topic-watchdog.ts
//
// D-09 / D-11 / D-12 — keep the topic issue wakeable.
//
// Single helper called from BOTH chat.send (D-12 per-message nudge) AND
// chat.messages on every poll (D-11 watchdog). PER RESEARCH OQ-1, hold
// the topic in 'in_progress' (non-terminal; chat-send.ts auto-reopen
// already flips to this value).
//
// FALSIFY-FIRST: the exact requestWakeup reason/contextSource strings
// the host's isVerifiedIssueTreeControlInteractionWake() accepts must
// be confirmed live on Countermoves (Plan 1 spike — see RESEARCH OQ-3).

import type { PluginIssuesClient, PluginLogger } from '@paperclipai/plugin-sdk';

const NON_TERMINAL_CONVERSATION_STATUS = 'in_progress'; // OQ-1 decision

export async function ensureTopicWakeable(
  ctx: { issues: PluginIssuesClient; logger?: PluginLogger },
  topicIssueId: string,
  companyId: string,
): Promise<void> {
  let issue: { status?: string } | null;
  try {
    issue = (await ctx.issues.get(topicIssueId, companyId)) as { status?: string } | null;
  } catch (e) {
    ctx.logger?.warn?.('topic-watchdog: issues.get failed', {
      topicIssueId, err: (e as Error).message,
    });
    return;
  }
  if (!issue) return;
  // D-09 / D-11 — topic must NEVER be terminal and must not be parked in
  // blocked recovery. Flip it back if observed wrong. Idempotent: a no-op
  // when status is already in_progress.
  if (issue.status === 'done' || issue.status === 'cancelled' || issue.status === 'blocked') {
    try {
      await ctx.issues.update(
        topicIssueId,
        { status: NON_TERMINAL_CONVERSATION_STATUS } as Parameters<PluginIssuesClient['update']>[1],
        companyId,
      );
    } catch (e) {
      ctx.logger?.warn?.('topic-watchdog: issues.update failed', {
        topicIssueId, err: (e as Error).message,
      });
    }
  }
  // D-12 — explicit per-message wake nudge. Bare comment SHOULD wake the
  // assignee natively (Phase 4 D-01), but multi-turn re-wake is unreliable
  // after a successful run (the observed CHT-1116 bug). requestWakeup is
  // the deterministic backstop. The capability issues.wakeup is already
  // declared (manifest.ts:297 — Plan 03-06).
  try {
    await ctx.issues.requestWakeup(topicIssueId, companyId, {
      reason: 'issue_commented',
      contextSource: 'clarity-pack:chat',
    });
  } catch (e) {
    // Best-effort — a wake gate rejection must not fail the caller.
    ctx.logger?.warn?.('topic-watchdog: requestWakeup failed', {
      topicIssueId, err: (e as Error).message,
    });
  }
}
```

**Stuck-topic signal helper** — same file should export `isTopicStuck(issue): boolean` reading `activeRecoveryAction` / `successfulRunHandoff` (UI-SPEC Pattern G trigger). The `Issue` shape carries these fields per `packages/shared/src/types/issue.ts` (RESEARCH sources).

### `src/worker/handlers/chat-active-tasks.ts` (NEW — data handler)

**Analog:** `src/worker/handlers/chat-topics.ts` `chat.topics` DATA handler (lines 101-134) — same wrap, same param-validation idiom, same map-row-to-entry shape. The query is different (`ctx.issues.list({originKind, …})` instead of a repo call).

**Pattern (adapt `chat-topics.ts:101-134`):**
```typescript
import { wrapDataHandler, type OptInGuardDataCtx } from '../opt-in-guard.ts';
import type { PluginIssuesClient, PluginLogger } from '@paperclipai/plugin-sdk';

export type ChatActiveTasksCtx = OptInGuardDataCtx & {
  issues: PluginIssuesClient;
  logger?: PluginLogger;
};

type ActiveTaskEntry = {
  issueId: string;
  identifier: string;   // BEAAA-NNN (issue.identifier from host)
  title: string;
  status: string;       // ISSUE_STATUSES value
  createdAt: string | null;
};

export function registerChatActiveTasks(ctx: ChatActiveTasksCtx): void {
  wrapDataHandler(ctx, 'chat.taskOwned', async (params) => {
    const companyId = typeof params?.companyId === 'string' && params.companyId ? params.companyId : null;
    const userId = typeof params?.userId === 'string' && params.userId ? params.userId : null;
    const topicIssueId = typeof params?.topicIssueId === 'string' && params.topicIssueId ? params.topicIssueId : null;
    if (!companyId) return { error: 'COMPANY_ID_REQUIRED' };
    if (!userId) return { error: 'USER_ID_REQUIRED' };
    if (!topicIssueId) return { error: 'TOPIC_ISSUE_ID_REQUIRED' };

    let issues: Array<{ id: string; identifier?: string; title?: string; status?: string; createdAt?: Date | string; originId?: string }>;
    try {
      // RESEARCH §"Don't Hand-Roll" — single ctx.issues.list call, NEVER per-task ctx.issues.get fan-out (Pitfall 5).
      issues = (await ctx.issues.list({
        companyId,
        originKind: 'plugin:clarity-pack',
      })) as Array<{ id: string; identifier?: string; title?: string; status?: string; createdAt?: Date | string; originId?: string }>;
    } catch (e) {
      ctx.logger?.warn?.('chat.taskOwned: list failed', { topicIssueId, err: (e as Error).message });
      return { error: 'TASKS_FAILED' };
    }

    const prefix = `chat-task:${topicIssueId}:`;
    const tasks: ActiveTaskEntry[] = (issues ?? [])
      .filter((i) => typeof i.originId === 'string' && i.originId.startsWith(prefix))
      .map((i) => ({
        issueId: i.id,
        identifier: i.identifier ?? i.id,
        title: i.title ?? '(untitled task)',
        status: i.status ?? 'todo',
        createdAt: i.createdAt instanceof Date ? i.createdAt.toISOString() : (typeof i.createdAt === 'string' ? i.createdAt : null),
      }));
    return { kind: 'taskOwned' as const, topicIssueId, tasks };
  });
}
```

If `ctx.issues.list({originKind})` does not return the `originId` field, fall back to a small `chat_topic_tasks` side table (no migration unless required). ### Opt-in-guard wrapping (CRITICAL — every worker handler)

**Source:** `src/worker/opt-in-guard.ts:87-100` + every chat handler

**Apply to:** `chat-true-task.ts`, `chat-topic-archive.ts`, `chat-active-tasks.ts` (new handlers), and the rewritten `chat-promote.ts`

**Excerpt** (the wrap call — same shape in every chat handler):
```typescript
// DATA handler:
wrapDataHandler(ctx, 'chat.taskOwned', async (params) => { … });

// ACTION handler:
wrapActionHandler(ctx, 'chat.createTrueTask', async (params) => { … });
```

The wrap automatically returns `{error: 'OPT_IN_REQUIRED'}` for opted-out callers BEFORE the inner handler runs — server-side enforcement under the same-origin trust model (T-04-15, OPTIN-04).

### `reqStr` param validation (every action handler)

**Source:** `chat-promote.ts:48-52` (also in `chat-topics.ts:76-80`, `chat-send.ts:44-48`)

**Apply to:** every action handler in this phase

```typescript
function reqStr(params: Record<string, unknown> | undefined, key: string): string {
  const v = params?.[key];
  if (typeof v === 'string' && v) return v;
  throw new Error(`<handler.key>: ${key} required`);
}
```

Action handlers THROW on missing required params (convention from research). Data handlers RETURN `{error: 'PARAM_REQUIRED'}` instead (see `chat-topics.ts:111-119` and `chat-messages.ts:95-103`).

### Worker-returned error pattern in UI

**Source:** `message-thread.tsx:449-455` (`PromoteActions.resultError`) + `composer.tsx:102-104`

**Apply to:** every new UI component that calls `usePluginAction` (`<TrueTaskDialog>`, `<ArchiveTopicButton>`, the wiring inside `<Composer>` for `chat.send`)

```typescript
function resultError(result: unknown): string | null {
  if (result && typeof result === 'object' && 'error' in result) {
    return String((result as { error: unknown }).error);
  }
  return null;
}
```

Action handlers RETURN structured `{error: …}` — they do NOT throw. UI MUST check the result, not rely on try/catch.

### usePluginData + usePoll pairing

**Source:** `message-thread.tsx:125-170` (the canonical pairing — initial fetch + 15s refresh)

**Apply to:** `<ActiveTasksOwned>` (new), and the existing `<MessageThread>` itself (no change to its poll mechanics)

```typescript
const { data, refresh } = usePluginData<Result>('chat.taskOwned', { … });
const REFRESH_INTERVAL_MS = 15_000;
usePoll<Result>({
```
